101 _id = -1 * id;
102 }
103

105 LIBSEMIGROUPS_ASSERT(_id != 0);
106 return _id;
107 }
108
109 void reorder() {
110 if (shortlex_compare(_lhs, _rhs)) {
111 std::swap(_lhs, _rhs);
112 }
113 }
