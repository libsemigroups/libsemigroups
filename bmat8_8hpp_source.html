704 [[nodiscard]] constexpr BMat8 transpose(BMat8 const& x) noexcept {
705 uint64_t y = x.to_int();
706 uint64_t z = (y ^ (y >> 7)) & 0xAA00AA00AA00AA;
707 y = y ^ z ^ (z << 7);
708 z = (y ^ (y >> 14)) & 0xCCCC0000CCCC;
709 y = y ^ z ^ (z << 14);
710 z = (y ^ (y >> 28)) & 0xF0F0F0F0;
711 y = y ^ z ^ (z << 28);
712 return BMat8(y);
713 }
