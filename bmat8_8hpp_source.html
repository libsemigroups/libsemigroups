155  inline BMat8 transpose() const {
156  uint64_t x = _data;
157  uint64_t y = (x ^ (x >> 7)) & 0xAA00AA00AA00AA;
158  x = x ^ y ^ (y << 7);
159  y = (x ^ (x >> 14)) & 0xCCCC0000CCCC;
160  x = x ^ y ^ (y << 14);
161  y = (x ^ (x >> 28)) & 0xF0F0F0F0;
162  x = x ^ y ^ (y << 28);
163  return BMat8(x);
164  }

172  inline BMat8 operator*(BMat8 const& that) const {
173  uint64_t y = that.transpose()._data;
174  uint64_t data = 0;
175  uint64_t tmp = 0;
176  uint64_t diag = 0x8040201008040201;
177  for (int i = 0; i < 8; ++i) {
178  tmp = _data & y;
179  tmp |= tmp >> 1;
180  tmp |= tmp >> 2;
181  tmp |= tmp >> 4;
182  tmp &= 0x0101010101010101;
183  tmp *= 255;
184  tmp &= diag;
185  data |= tmp;
186  y = cyclic_shift(y);
187  diag = cyclic_shift(diag);
188  }
189  return BMat8(data);
190  }

260  // Cyclically shifts bits to left by 8m
261  // https://stackoverflow.com/a/776523
262  static inline uint64_t cyclic_shift(uint64_t n, uint64_t m = 1) {
263  const unsigned int mask
264  = (CHAR_BIT * sizeof(n) - 1); // assumes width is a power of 2.
265 
266  // assert ( (c<=mask) &&"rotate by type width or more");
267  unsigned int c = 8 * m;
268  c &= mask;
269  return (n << c) | (n >> ((-c) & mask));
270  }
