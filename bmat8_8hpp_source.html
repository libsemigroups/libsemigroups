710 [[nodiscard]] constexpr BMat8 transpose(BMat8 const& x) noexcept {
711 uint64_t y = x.to_int();
712 uint64_t z = (y ^ (y >> 7)) & 0xAA00AA00AA00AA;
713 y = y ^ z ^ (z << 7);
714 z = (y ^ (y >> 14)) & 0xCCCC0000CCCC;
715 y = y ^ z ^ (z << 14);
716 z = (y ^ (y >> 28)) & 0xF0F0F0F0;
717 y = y ^ z ^ (z << 28);
718 return BMat8(y);
719 }
