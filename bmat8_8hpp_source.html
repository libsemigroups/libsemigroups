719 [[nodiscard]] constexpr BMat8 transpose(BMat8 const& x) noexcept {
720 uint64_t y = x.to_int();
721 uint64_t z = (y ^ (y >> 7)) & 0xAA00AA00AA00AA;
722 y = y ^ z ^ (z << 7);
723 z = (y ^ (y >> 14)) & 0xCCCC0000CCCC;
724 y = y ^ z ^ (z << 14);
725 z = (y ^ (y >> 28)) & 0xF0F0F0F0;
726 y = y ^ z ^ (z << 28);
727 return BMat8(y);
728 }
