2057 template <typename Word = std::string>
2058 typename Word::value_type human_readable_letter(size_t i) {
2059 // This check ensures that i is not too large to be converted to a
2060 // Word::value_type. This is check is only needed if the number of
2061 // distinct Word::value objects is less than the number of distinct size_t
2062 // objects.
2063 if constexpr (sizeof(typename Word::value_type) < sizeof(size_t)) {
2064 if (i > std::numeric_limits<typename Word::value_type>::max()
2065 - std::numeric_limits<typename Word::value_type>::min()) {
2066 LIBSEMIGROUPS_EXCEPTION(
2067 "expected the argument to be in the range [0, {}), found {}",
2068 1 + std::numeric_limits<typename Word::value_type>::max()
2069 - std::numeric_limits<typename Word::value_type>::min(),
2070 i);
2071 }
2072 }
2073 if constexpr (!std::is_same_v<Word, std::string>) {
2074 return static_cast<typename Word::value_type>(i);
2075 } else {
2076 // Choose visible characters a-zA-Z0-9 first before anything else
2077 // The ascii ranges for these characters are: [97, 123), [65, 91),
2078 // [48, 58) so the remaining range of chars that are appended to the end
2079 // after these chars are [0,48), [58, 65), [91, 97), [123, 255]
2080 return detail::chars_in_human_readable_order()[i];
2081 }
2082 }

2338 void pow_inplace(Word& x, size_t n) {
2339 Word y(x);
2340 x.reserve(x.size() * n);
2341 if (n % 2 == 0) {
2342 x = Word({});
2343 }
2344
2345 while (n > 1) {
2346 y += y;
2347 n /= 2;
2348 if (n % 2 == 1) {
2349 x += y;
2350 }
2351 }
2352 }

2358 // Note: we could do a version of the below using insert on words, where
2359 // the step is +/- 1.
2360 template <typename Container, typename Word>
2361 Word prod(Container const& elts, int first, int last, int step) {
2362 if (step == 0) {
2363 LIBSEMIGROUPS_EXCEPTION("the 4th argument must not be 0");
2364 } else if (((first < last && step > 0) || (first > last && step < 0))
2365 && elts.size() == 0) {
2366 LIBSEMIGROUPS_EXCEPTION("the 1st argument must not be empty if the "
2367 "given range is not empty");
2368 // TODO Is int signed? Should this also contain
2369 // std::numeric_limits<int>::min?
2370 } else if (elts.size() > std::numeric_limits<int>::max()) {
2371 LIBSEMIGROUPS_EXCEPTION(
2372 "the 1st argument must have size less than or equal to {}",
2373 std::numeric_limits<int>::max());
2374 }
2375 Word result;
2376 int const s = elts.size();
2377
2378 if (first < last) {
2379 if (step < 0) {
2380 return result;
2381 }
2382 result.reserve((last - first) / step);
2383
2384 for (int i = first; i < last; i += step) {
2385 size_t const a = (i % s + s) % s;
2386 LIBSEMIGROUPS_ASSERT(static_cast<int>(a) < s);
2387 result += elts[a];
2388 }
2389 } else {
2390 if (step > 0) {
2391 return result;
2392 }
2393 size_t steppos = static_cast<size_t>(-step);
2394 result.reserve((first - last) / steppos);
2395 for (int i = first; i > last; i += step) {
2396 size_t const a = (i % s + s) % s;
2397 LIBSEMIGROUPS_ASSERT(static_cast<int>(a) < s);
2398 result += elts[a];
2399 }
2400 }
2401 return result;
2402 }
