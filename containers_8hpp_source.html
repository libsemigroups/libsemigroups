43 template <typename T, size_t N>
44 class StaticTriVector2 {
45 // So that StaticTriVector2<T, N> can access private data members
46 // of StaticTriVector2<S, M> and vice versa.
47 template <typename S, size_t M>
48 friend class StaticTriVector2;
49
50 public:
51 // Aliases for iterators
52 using const_iterator =
53 typename std::array<T, N*(N + 1) / 2>::const_iterator;
54
55 StaticTriVector2(StaticTriVector2 const&) = default;
56 StaticTriVector2(StaticTriVector2&&) = default;
57 StaticTriVector2& operator=(StaticTriVector2 const&) = default;
58 StaticTriVector2& operator=(StaticTriVector2&&) = default;
59 ~StaticTriVector2() = default;
60
61 StaticTriVector2() : _data(), _sizes() {
62 clear();
63 }
64
65 // Not noexcept since std::array::fill is not.
66 void clear() {
67 _sizes.fill(0);
68 }
69
70 // Not noexcept because std::array::operator[] isn't
71 void push_back(size_t depth, T x) {
72 LIBSEMIGROUPS_ASSERT(depth < N);
73 LIBSEMIGROUPS_ASSERT(_sizes[depth] < N - depth);
74 _data[depth * (2 * N - depth + 1) / 2 + _sizes[depth]] = x;
75 _sizes[depth]++;
76 }
77
78 // Not noexcept because std::array::operator[] isn't
79 inline T back(size_t depth) const {
80 LIBSEMIGROUPS_ASSERT(depth < N);
81 return _data[depth * (2 * N - depth + 1) / 2 + _sizes[depth] - 1];
82 }
83
84 // Not noexcept because std::array::operator[] isn't
85 inline T const& at(size_t depth, size_t index) const {
86 LIBSEMIGROUPS_ASSERT(depth < N);
87 LIBSEMIGROUPS_ASSERT(index < _sizes[depth]);
88 return _data[depth * (2 * N - depth + 1) / 2 + index];
89 }
90
91 // Not noexcept because std::array::operator[] isn't
92 inline size_t size(size_t depth) const {
93 LIBSEMIGROUPS_ASSERT(depth < N);
94 return _sizes[depth];
95 }
96
97 inline const_iterator cbegin(size_t depth) const noexcept {
98 LIBSEMIGROUPS_ASSERT(depth < N);
99 return _data.cbegin() + depth * (2 * N - depth + 1) / 2;
100 }
101
102 // Not noexcept because std::array::operator[] isn't
103 inline const_iterator cend(size_t depth) const {
104 LIBSEMIGROUPS_ASSERT(depth < N);
105 return _data.cbegin() + depth * (2 * N - depth + 1) / 2 + _sizes[depth];
106 }
107
108 inline const_iterator begin(size_t depth) const noexcept {
109 LIBSEMIGROUPS_ASSERT(depth < N);
110 return _data.begin() + depth * (2 * N - depth + 1) / 2;
111 }
112
113 // Not noexcept because std::array::operator[] isn't
114 inline const_iterator end(size_t depth) const {
115 LIBSEMIGROUPS_ASSERT(depth < N);
116 return _data.begin() + depth * (2 * N - depth + 1) / 2 + _sizes[depth];
117 }
118
119 inline size_t max_size(size_t depth) const noexcept {
120 return N - depth;
121 }
122
123 [[nodiscard]] bool empty() const noexcept {
124 return std::count(_sizes.cbegin(), _sizes.cend(), 0) == N;
125 }
126
127 private:
128 std::array<T, N*(N + 1) / 2> _data;
129 std::array<size_t, N> _sizes;
130 };

132 // Template class for 2-dimensional dynamic arrays.
133 template <typename T, typename A = std::allocator<T>>
134 class DynamicArray2 {
135 // So that DynamicArray2<T> can access private data members of
136 // DynamicArray2<S> and vice versa.
137 template <typename S, typename B>
138 friend class DynamicArray2;
139
140 public:
141 using allocator_type = A;
142 using value_type = typename std::vector<T, A>::value_type;
143 using reference = typename std::vector<T, A>::reference;
144 using const_reference = typename std::vector<T, A>::const_reference;
145 using difference_type = typename std::vector<T, A>::difference_type;
146 using size_type = typename std::vector<T, A>::size_type;
147 using const_pointer = typename std::vector<T, A>::const_pointer;
148 using pointer = typename std::vector<T, A>::pointer;
149
150 DynamicArray2(DynamicArray2 const&);
151 DynamicArray2(DynamicArray2&&);
152 DynamicArray2& operator=(DynamicArray2 const&);
153 DynamicArray2& operator=(DynamicArray2&&);
154
155 ~DynamicArray2() = default;
156
157 // Not noexcept because DynamicArray2::add_rows can throw.
158 explicit DynamicArray2(size_type number_of_cols = 0,
159 size_type number_of_rows = 0,
160 T default_val = 0)
161 : _vec(),
162 _nr_used_cols(number_of_cols),
163 _nr_unused_cols(0),
164 _nr_rows(0),
165 _default_val(default_val) {
166 this->add_rows(number_of_rows);
167 }
168
169 DynamicArray2& init(size_type number_of_cols,
170 size_type number_of_rows,
171 value_type default_val) {
172 _default_val = default_val;
173 reshape(number_of_cols, number_of_rows);
174 return *this;
175 }
176
177 // Not noexcept because DynamicArray2::DynamicArray2(size_type, size_type)
178 // can throw.
179 explicit DynamicArray2(std::initializer_list<std::initializer_list<T>> il)
180 : DynamicArray2(il.begin()->size(), il.size()) {
181 auto it = _vec.begin();
182 for (auto const& row : il) {
183 LIBSEMIGROUPS_ASSERT(row.size() == _nr_used_cols);
184 std::copy(row.begin(), row.end(), it);
185 it += _nr_used_cols + _nr_unused_cols;
186 }
187 }
188
189 // Not noexcept
190 template <typename S, typename B>
191 DynamicArray2(DynamicArray2<S, B> const& copy,
192 size_type number_of_cols_to_add = 0)
193 : _vec(),
194 _nr_used_cols(copy._nr_used_cols),
195 _nr_unused_cols(copy._nr_unused_cols),
196 _nr_rows(copy.number_of_rows()),
197 _default_val(copy._default_val) {
198 if (number_of_cols_to_add <= _nr_unused_cols) {
199 _vec.assign(copy._vec.cbegin(), copy._vec.cend());
200 _nr_used_cols += number_of_cols_to_add;
201 _nr_unused_cols -= number_of_cols_to_add;
202 return;
203 }
204
205 size_type new_nr_cols = std::max(
206 2 * number_of_cols(), number_of_cols_to_add + number_of_cols());
207 _nr_used_cols += number_of_cols_to_add;
208 _nr_unused_cols = new_nr_cols - _nr_used_cols;
209
210 _vec.reserve(new_nr_cols * _nr_rows);
211 // TODO(later) improve
212 for (size_type i = 0; i < _nr_rows; i++) {
213 size_type j;
214 for (j = 0; j < copy._nr_used_cols; j++) {
215 _vec.push_back(copy.get(i, j));
216 }
217 for (; j < new_nr_cols; j++) {
218 _vec.push_back(_default_val);
219 }
220 }
221 }

223 // Not noexcept, since std::equal can throw
224 bool operator==(DynamicArray2<T, A> const& that) const {
225 if (_nr_used_cols != that._nr_used_cols || _nr_rows != that._nr_rows) {
226 return false;
227 }
228 return std::equal(cbegin(), cend(), that.cbegin());
229 }
230
231 bool operator<(DynamicArray2<T, A> const& that) const {
232 return std::lexicographical_compare(
233 _vec.cbegin(), _vec.cend(), that.cbegin(), that.cend());
234 }
235
236 bool operator>(DynamicArray2<T, A> const& that) const {
237 return that < *this;
238 }
239
240 // Not noexcept, since operator== can throw
241 bool operator!=(DynamicArray2<T, A> const& that) const {
242 return !operator==(that);
243 }
244
245 bool operator<=(DynamicArray2<T, A> const& that) const {
246 return *this < that || *this == that;
247 }
248
249 bool operator>=(DynamicArray2<T, A> const& that) const {
250 return *this > that || *this == that;
251 }
252
253 bool empty() const noexcept {
254 return _nr_rows == 0;
255 }
256
257 size_type size() const noexcept {
258 return _nr_rows * _nr_used_cols;
259 }
260
261 size_type max_size() const noexcept {
262 return _vec.max_size();
263 }
264
265 // Not noexcept, since std::fill can throw
266 void fill(T const& val) {
267 std::fill(_vec.begin(), _vec.end(), val);
268 }
269
270 void swap(DynamicArray2<T, A>& that) noexcept {
271 _vec.swap(that._vec);
272 std::swap(_nr_used_cols, that._nr_used_cols);
273 std::swap(_nr_unused_cols, that._nr_unused_cols);
274 std::swap(_nr_rows, that._nr_rows);
275 }

277 // Not noexcept because std::vector::operator[] isn't
278 void swap(size_type i, size_type j, size_type k, size_type l) {
279 LIBSEMIGROUPS_ASSERT(i < _nr_rows);
280 LIBSEMIGROUPS_ASSERT(j < _nr_used_cols);
281 LIBSEMIGROUPS_ASSERT(k < _nr_rows);
282 LIBSEMIGROUPS_ASSERT(l < _nr_used_cols);
283 std::swap(_vec[i * (_nr_used_cols + _nr_unused_cols) + j],
284 _vec[k * (_nr_used_cols + _nr_unused_cols) + l]);
285 }
286
287 // Not noexcept since std::swap_ranges can throw.
288 void swap_rows(size_type i, size_type j) {
289 LIBSEMIGROUPS_ASSERT(i < _nr_rows);
290 LIBSEMIGROUPS_ASSERT(j < _nr_rows);
291 size_type const number_of_cols = _nr_used_cols + _nr_unused_cols;
292 std::swap_ranges(_vec.begin() + (i * number_of_cols),
293 _vec.begin() + ((i + 1) * number_of_cols),
294 _vec.begin() + (j * number_of_cols));
295 }

297 // The following is adapted from http://bit.ly/2X4xPlK
298 // TODO(later) 1. make this a non-member function
299 // 2. should perform checks that p actually permutes the
300 // given row
301 // Not noexcept because std::vector::operator[] isn't
302 void apply_row_permutation(std::vector<T> p) {
303 for (size_type i = 0; i < p.size(); i++) {
304 size_type current = i;
305 while (i != p[current]) {
306 size_type next = p[current];
307 swap_rows(current, next);
308 p[current] = current;
309 current = next;
310 }
311 p[current] = current;
312 }
313 }

316 _nr_unused_cols += _nr_used_cols;

325 clear();

330 // Throws if the assignment operator of T throws
331 void shrink_rows_to(size_type n) {
332 if (n < _nr_rows) {
333 shrink_rows_to_no_checks(0, n);
334 }
335 }
336
337 void shrink_rows_to_no_checks(size_type first, size_type last) {
338 LIBSEMIGROUPS_ASSERT(first <= last);
339 LIBSEMIGROUPS_ASSERT(first <= _nr_rows);
340 LIBSEMIGROUPS_ASSERT(last <= _nr_rows);
341
342 auto nr_cols = _nr_used_cols + _nr_unused_cols;
343 _vec.erase(_vec.begin() + last * nr_cols, _vec.end());
344 _vec.erase(_vec.begin(), _vec.begin() + first * nr_cols);
345 _vec.shrink_to_fit();
346 _nr_rows = last - first;
347 }

349 // Throws if the assignment operator of T throws
350 void erase_column(size_type i) {
351 LIBSEMIGROUPS_ASSERT(i < _nr_used_cols);
352 size_type const n = _nr_used_cols + _nr_unused_cols;
353 for (auto it = _vec.rbegin() + (n - i); it < _vec.rend(); it += n) {
354 _vec.erase(it.base());
355 }
356 _nr_used_cols--;
357 }

359 // Not noexcept since std::vector::resize can throw.
360 void inline add_rows(size_type nr) {
361 _nr_rows += nr;
362 if (nr != 0) {
363 _vec.resize(_vec.size() + (_nr_used_cols + _nr_unused_cols) * nr,
364 _default_val);
365 }
366 }

368 void inline set_default_value(T val) noexcept {
369 _default_val = val;

376 // Not noexcept
377 void add_cols(size_type nr) {
378 if (nr <= _nr_unused_cols) {
379 _nr_used_cols += nr;
380 _nr_unused_cols -= nr;
381 return;
382 }
383
384 size_type old_nr_cols = _nr_used_cols + _nr_unused_cols;
385 size_type new_nr_cols = std::max(2 * old_nr_cols, nr + old_nr_cols);
386 if (_nr_rows != 0) {
387 _vec.resize(new_nr_cols * _nr_rows, _default_val);
388
389 typename std::vector<T>::iterator old_it(
390 _vec.begin() + (old_nr_cols * _nr_rows) - old_nr_cols);
391 typename std::vector<T>::iterator new_it(
392 _vec.begin() + (new_nr_cols * _nr_rows) - new_nr_cols);
393
394 while (old_it != _vec.begin()) {
395 std::move(old_it, old_it + _nr_used_cols, new_it);
396 old_it -= old_nr_cols;
397 new_it -= new_nr_cols;
398 }
399 }
400 _nr_used_cols += nr;
401 _nr_unused_cols = new_nr_cols - _nr_used_cols;
402 }

405 void set(size_type i, size_type j, T val) {
406 LIBSEMIGROUPS_ASSERT(i < _nr_rows && j < _nr_used_cols);
407 _vec[i * (_nr_used_cols + _nr_unused_cols) + j] = val;
408 }

424 // Not noexcept
425 template <typename S, typename B>
426 void append(DynamicArray2<S, B> const& copy) {
427 LIBSEMIGROUPS_ASSERT(copy._nr_used_cols == _nr_used_cols);
428
429 size_type old_nr_rows = _nr_rows;
430 add_rows(copy._nr_rows);
431
432 if (copy._nr_unused_cols == _nr_unused_cols) {
433 std::copy(copy._vec.begin(),
434 copy._vec.end(),
435 _vec.begin()
436 + (_nr_used_cols + _nr_unused_cols) * old_nr_rows);
437 } else { // TODO(later) improve this
438 for (size_type i = old_nr_rows; i < _nr_rows; i++) {
439 for (size_type j = 0; j < _nr_used_cols; j++) {
440 set(i, j, copy.get(i - old_nr_rows, j));
441 }
442 }
443 }
444 }

448 _vec.reserve(number_of_rows * (_nr_unused_cols + _nr_used_cols));
449 }

458 // Helper functions for iterators
459 static inline size_type remainder(difference_type a,
460 difference_type b) noexcept {
461 if (a >= 0) {
462 return a % b;
463 } else {
464 return (a % b) + b;
465 }
466 }
467
468 static inline size_type quotient(difference_type a,
469 difference_type b) noexcept {
470 LIBSEMIGROUPS_ASSERT(b != 0);
471 if (a % b >= 0) {
472 return a / b;
473 } else {
474 return (a / b) - 1;
475 }
476 }

478 // Types for ConstIteratorStateful
479 template <typename TInternalIteratorType>
480 friend struct AddAssign;
481
482 template <typename TInternalIteratorType>
483 struct AddAssign {
484 TInternalIteratorType& operator()(DynamicArray2 const* da,
485 TInternalIteratorType& it,
486 size_type val) const noexcept {
487 if (da->_nr_unused_cols == 0 || val == 0) {
488 return it += val;
489 }
490 size_type s = da->number_of_cols();
491 size_type n = da->_nr_unused_cols;
492 LIBSEMIGROUPS_ASSERT(s != 0);
493 size_type q = val / s;
494 size_type r = val % s;
495
496 difference_type p = it - da->_vec.cbegin();
497 it += q * (s + n) + r;
498
499 if (r + remainder(p, s + n) >= s) {
500 it += n;
501 }
502 return it;
503 }
504 };
505
506 template <typename TInternalIteratorType>
507 friend struct SubtractAssign;
508
509 template <typename TInternalIteratorType>
510 struct SubtractAssign {
511 TInternalIteratorType& operator()(DynamicArray2 const* da,
512 TInternalIteratorType& it,
513 size_type val) const noexcept {
514 if (da->_nr_unused_cols == 0 || val == 0) {
515 return it -= val;
516 }
517 size_type s = da->number_of_cols();
518 size_type n = da->_nr_unused_cols;
519 LIBSEMIGROUPS_ASSERT(s != 0);
520 size_type q = val / s;
521 size_type r = val % s;
522
523 difference_type p = it - da->_vec.cbegin();
524 it -= (q * (s + n) + r);
525 if (remainder(p, s + n) < r) {
526 it -= n;
527 }
528 return it;
529 }
530 };

553 template <typename TInternalIteratorType>
554 struct Difference {
555 inline difference_type
556 operator()(DynamicArray2 const* da1,
557 TInternalIteratorType const& it1,
558 DynamicArray2 const* da2,
559 TInternalIteratorType const& it2) const noexcept {
560 LIBSEMIGROUPS_ASSERT(da1 == da2);
561 (void) da2;
562 difference_type s = da1->_nr_used_cols;
563 difference_type n = da1->_nr_unused_cols;
564 difference_type b = it1 - da1->_vec.begin();
565 difference_type a = it2 - da1->_vec.begin();
566 if (s == 0 && n == 0) {
567 return 0;
568 }
569 return (b - a) - n * (quotient(b, s + n) - quotient(a, s + n));
570 }
571 };
