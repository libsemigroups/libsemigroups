674  template <typename T>
675  class PartialPerm : public PartialTransformation<T, PartialPerm<T>> {
676  public:
677  using PartialTransformation<T, PartialPerm<T>>::PartialTransformation;
678  using PartialTransformation<T, PartialPerm<T>>::UNDEFINED;
679 
687  explicit PartialPerm(std::vector<T> const& dom,
688  std::vector<T> const& ran,
689  size_t deg)
690  : PartialTransformation<T, PartialPerm<T>>() {
691  LIBSEMIGROUPS_ASSERT(dom.size() == ran.size());
692  LIBSEMIGROUPS_ASSERT(dom.empty()
693  || deg >= *std::max_element(dom.begin(), dom.end()));
694 
695  this->_vector->resize(deg + 1, UNDEFINED);
696  for (size_t i = 0; i < dom.size(); i++) {
697  (*this->_vector)[dom[i]] = ran[i];
698  }
699  }
700 
708  bool operator<(const Element& that) const override {
709  auto pp_this = static_cast<const PartialPerm<T>*>(this);
710  auto pp_that = static_cast<const PartialPerm<T>&>(that);
711 
712  size_t deg_this = pp_this->degree();
713  for (auto it = pp_this->_vector->end() - 1;
714  it >= pp_this->_vector->begin();
715  it--) {
716  if (*it == UNDEFINED) {
717  deg_this--;
718  } else {
719  break;
720  }
721  }
722  size_t deg_that = pp_that.degree();
723  for (auto it = pp_that._vector->end() - 1;
724  it >= pp_that._vector->begin() && deg_that >= deg_this;
725  it--) {
726  if (*it == UNDEFINED) {
727  deg_that--;
728  } else {
729  break;
730  }
731  }
732 
733  if (deg_this != deg_that) {
734  return deg_this < deg_that;
735  }
736 
737  for (size_t i = 0; i < deg_this; i++) {
738  if ((*pp_this)[i] != pp_that[i]) {
739  return (*pp_this)[i] == UNDEFINED
740  || (pp_that[i] != UNDEFINED && (*pp_this)[i] < pp_that[i]);
741  }
742  }
743  return false;
744  }
745 
752  Element* really_copy(size_t increase_deg_by = 0) const override {
753  PartialPerm<T>* copy
754  = new PartialPerm<T>(new std::vector<T>(*this->_vector));
755  if (increase_deg_by == 0) {
756  copy->_hash_value = this->_hash_value;
757  } else {
758  size_t n = copy->_vector->size();
759  copy->_vector->reserve(n + increase_deg_by);
760  for (size_t i = n; i < n + increase_deg_by; i++) {
761  copy->_vector->push_back(UNDEFINED);
762  }
763  }
764  return copy;
765  }

773  void redefine(Element const* x, Element const* y) override {
774  LIBSEMIGROUPS_ASSERT(x->degree() == y->degree());
775  LIBSEMIGROUPS_ASSERT(x->degree() == this->degree());
776  LIBSEMIGROUPS_ASSERT(x != this && y != this);

1139  void redefine(Element const* x, Element const* y) override {
1140  auto xx = static_cast<MatrixOverSemiringBase const*>(x);
1141  auto yy = static_cast<MatrixOverSemiringBase const*>(y);
1142  LIBSEMIGROUPS_ASSERT(xx->degree() == yy->degree());
1143  LIBSEMIGROUPS_ASSERT(xx->degree() == this->degree());
1144  LIBSEMIGROUPS_ASSERT(xx != this && yy != this);
1145  // It can be that the elements are defined over semirings that are
1146  // distinct in memory but equal (for example, when one element comes
1147  // from a semigroup and another from an ideal of that semigroup).
1148  // LIBSEMIGROUPS_ASSERT(xx->semiring() == yy->semiring() &&
1149  // xx->semiring() == this->semiring());
1150  // TODO verify that x, y, and this are defined over the same semiring.
1151  size_t deg = this->degree();
1152 
1153  for (size_t i = 0; i < deg; i++) {
1154  for (size_t j = 0; j < deg; j++) {
1155  int64_t v = _semiring->zero();
1156  for (size_t k = 0; k < deg; k++) {
1157  v = _semiring->plus(
1158  v, _semiring->prod((*xx)[i * deg + k], (*yy)[k * deg + j]));
1159  }
1160  (*this->_vector)[i * deg + j] = v;
1161  }
1162  }
1163  after(); // post process this
1164  this->reset_hash_value();
1165  }
