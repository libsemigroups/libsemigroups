1016 template <typename T>
1017 void add_generators(T const& first, T const& last) {
1018 if (started()) {
1019 LIBSEMIGROUPS_EXCEPTION(
1020 "cannot add generators after the algorithm has begun!");
1021 }
1022 validate_element_collection(first, last);
1023 // If the _one has already been pushed back into _gens
1024 // (i.e. if data has been initialised), then we should shuffle it to
1025 // the back of the new generators.
1026 if (_data_initialised) {
1027 _gens.pop_back();
1028 }
1029 for (auto it = first; it < last; ++it) {
1030 _gens.push_back(this->internal_copy(this->to_internal_const(*it)));
1031 }
1032 if (_data_initialised) {
1033 _gens.push_back(_one);
1034 }
1035 // Once generators are added, the data structures can be initialised.
1036 // init_data() does nothing if it has already been run.
1037 init_data();
1038 // However, the next line does always do something; the rank state and rep
1039 // vecs depend on the generators.
1040 init_rank_state_and_rep_vecs();
1041 }

1269 // Returns a lambda orb index corresponding to a group H-class in the R-
1270 // class of \p x.
1271 // asserts its argument has lambda/rho values in the orbits.
1272 // modifies _tmp_lambda_value1
1273 // modifies _tmp_rho_value1
1274 lambda_orb_index_type get_lambda_group_index(internal_const_reference x) {
1275 auto lpos = get_lpos(x);
1276 LIBSEMIGROUPS_ASSERT(lpos != UNDEFINED);
1277
1278 auto lval_scc_id = _lambda_orb.scc().id(lpos);
1279
1280 std::pair key(get_rpos(x), lval_scc_id);
1281
1282 if (_group_indices.find(key) != _group_indices.end()) {
1283 return _group_indices.at(key);
1284 }
1285
1286 PoolGuard cg1(_element_pool);
1287 PoolGuard cg2(_element_pool);
1288 internal_element_type tmp1 = cg1.get();
1289 internal_element_type tmp2 = cg2.get();
1290
1291 Product()(this->to_external(tmp1),
1292 this->to_external_const(x),
1293 _lambda_orb.multiplier_to_scc_root(lpos));
1294
1295 for (auto const& val : _lambda_orb.scc().component(lval_scc_id)) {
1296 Product()(this->to_external(tmp2),
1297 this->to_external(tmp1),
1298 _lambda_orb.multiplier_from_scc_root(val));
1299 if (is_group_index(x, tmp2)) {
1300 _group_indices.emplace(key, val);
1301 return val;
1302 }
1303 }
1304 _group_indices.emplace(key, UNDEFINED);
1305 return UNDEFINED;
1306 }
