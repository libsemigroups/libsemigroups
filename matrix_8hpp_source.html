498 // not noexcept because memory is allocated
499 void product_inplace_no_checks(Subclass const& A, Subclass const& B) {
500 LIBSEMIGROUPS_ASSERT(number_of_rows() == number_of_cols());
501 LIBSEMIGROUPS_ASSERT(A.number_of_rows() == number_of_rows());
502 LIBSEMIGROUPS_ASSERT(B.number_of_rows() == number_of_rows());
503 LIBSEMIGROUPS_ASSERT(A.number_of_cols() == number_of_cols());
504 LIBSEMIGROUPS_ASSERT(B.number_of_cols() == number_of_cols());
505 LIBSEMIGROUPS_ASSERT(&A != this);
506 LIBSEMIGROUPS_ASSERT(&B != this);
507
508 // Benchmarking boolean matrix multiplication reveals that using a
509 // non-static container_type gives the best performance, when compared
510 // to static container_type the performance is more or less the same
511 // (but not thread-safe), and there appears to be a performance
512 // penalty of about 50% when using static thread_local container_type
513 // (when compiling with clang).
514 size_t const N = A.number_of_rows();
515 std::vector<scalar_type> tmp(N, 0);
516
517 for (size_t c = 0; c < N; c++) {
518 for (size_t i = 0; i < N; i++) {
519 tmp[i] = B(i, c);
520 }
521 for (size_t r = 0; r < N; r++) {
522 (*this)(r, c) = std::inner_product(
523 A._container.begin() + r * N,
524 A._container.begin() + (r + 1) * N,
525 tmp.begin(),
526 scalar_zero(),
527 [this](scalar_type x, scalar_type y) {
528 return this->plus_no_checks(x, y);
529 },
530 [this](scalar_type x, scalar_type y) {
531 return this->product_no_checks(x, y);
532 });
533 }
534 }
535 }

7396 // This works with std::vector and StaticVector1, with value_type equal
7397 // to std::bitset and BitSet.
7398 template <typename Mat, typename Container>
7399 void bitset_row_basis(Container&& rows, std::decay_t<Container>& result) {
7400 using value_type = typename std::decay_t<Container>::value_type;
7401 static_assert(IsBMat<Mat>, "IsBMat<Mat> must be true!");
7402 static_assert(IsBitSet<value_type> || detail::IsStdBitSet<value_type>,

7406 || rows[0].size() <= BitSet<1>::max_size());
7407
7408 std::sort(rows.begin(), rows.end(), detail::LessBitSet());
7409 // Remove duplicates
7410 rows.erase(std::unique(rows.begin(), rows.end()), rows.end());
7411 for (size_t i = 0; i < rows.size(); ++i) {
7412 value_type cup;
7413 cup.reset();
7414 for (size_t j = 0; j < i; ++j) {
7415 if ((rows[i] & rows[j]) == rows[j]) {
7416 cup |= rows[j];
7417 }
7418 }
7419 for (size_t j = i + 1; j < rows.size(); ++j) {
7420 if ((rows[i] & rows[j]) == rows[j]) {
7421 cup |= rows[j];
7422 }
7423 }
7424 if (cup != rows[i]) {
7425 result.push_back(std::move(rows[i]));
7426 }
7427 }

7567 //! \f$O(r ^ 2 c)\f$ where \f$r\f$ is the size of \p views and
7568 //! \f$c\f$ is the size of each row view or bit set in \p views.
7569 template <typename Mat, typename Container>
7570 std::enable_if_t<IsMaxPlusTruncMat<Mat>>
7571 row_basis(Container&& views, std::decay_t<Container>& result) {
7572 using value_type = typename std::decay_t<Container>::value_type;
7573 static_assert(std::is_same<value_type, typename Mat::RowView>::value,
7574 "Container::value_type must be Mat::RowView");
7575 using scalar_type = typename Mat::scalar_type;
7576 using Row = typename Mat::Row;
7577
7578 if (views.empty()) {
7579 return;
7580 }
7581
7582 LIBSEMIGROUPS_ASSERT(result.empty());
7583
7584 std::sort(views.begin(), views.end());
7585 Row tmp1(views[0]);
7586
7587 for (size_t r1 = 0; r1 < views.size(); ++r1) {
7588 if (r1 == 0 || views[r1] != views[r1 - 1]) {
7589 std::fill(tmp1.begin(), tmp1.end(), tmp1.scalar_zero());
7590 for (size_t r2 = 0; r2 < r1; ++r2) {
7591 scalar_type max_scalar = matrix::threshold(tmp1);
7592 for (size_t c = 0; c < tmp1.number_of_cols(); ++c) {
7593 if (views[r2][c] == tmp1.scalar_zero()) {
7594 continue;
7595 }
7596 if (views[r1][c] >= views[r2][c]) {
7597 if (views[r1][c] != matrix::threshold(tmp1)) {
7598 max_scalar
7599 = std::min(max_scalar, views[r1][c] - views[r2][c]);
7600 }
7601 } else {
7602 max_scalar = tmp1.scalar_zero();
7603 break;
7604 }
7605 }
7606 if (max_scalar != tmp1.scalar_zero()) {
7607 tmp1 += views[r2] * max_scalar;
7608 }
7609 }
7610 if (tmp1 != views[r1]) {
7611 result.push_back(views[r1]);
7612 }
7613 }
7614 }
7615 }

8263 //!
8264 //! \throws LibsemigroupsException if \p braces does not have size \c 2.
8265 template <typename Mat>
8266 auto to_human_readable_repr(Mat const& x,
8267 std::string const& prefix,
8268 std::string const& short_name = "",
8269 std::string const& braces = "{}",
8270 size_t max_width = 72)
8271 -> std::enable_if_t<IsMatrix<Mat>, std::string> {
8272 if (braces.size() != 2) {
8273 LIBSEMIGROUPS_EXCEPTION(
8274 "the 4th argument (braces) must have size 2, found {}",
8275 braces.size());
8276 }
8277
8278 size_t const R = x.number_of_rows();
8279 size_t const C = x.number_of_cols();
8280
8281 std::vector<size_t> max_col_widths(C, 0);
8282 std::vector<size_t> row_widths(C, prefix.size() + 1);
8283 for (size_t r = 0; r < R; ++r) {
8284 for (size_t c = 0; c < C; ++c) {
8285 size_t width
8286 = detail::unicode_string_length(detail::entry_repr(x(r, c)));
8287 row_widths[r] += width;
8288 if (width > max_col_widths[c]) {
8289 max_col_widths[c] = width;
8290 }
8291 }
8292 }
8293 auto col_width
8294 = *std::max_element(max_col_widths.begin(), max_col_widths.end());
8295 // The total width if we pad the entries according to the widest column.
8296 auto const total_width = col_width * C + prefix.size() + 1;
8297 if (total_width > max_width) {
8298 // Padding according to the widest column is too wide!
8299 if (*std::max_element(row_widths.begin(), row_widths.end()) > max_width) {
8300 // If the widest row is too wide, then use the short name
8301 return fmt::format(
8302 "<{}x{} {}>", x.number_of_rows(), x.number_of_cols(), short_name);
8303 }
8304 // If the widest row is not too wide, then just don't pad the entries
8305 col_width = 0;
8306 }
8307
8308 std::string result = fmt::format("{}", prefix);
8309 std::string rindent;
8310 auto const lbrace = braces[0], rbrace = braces[1];
8311 if (R != 0 && C != 0) {
8312 result += lbrace;
8313 for (size_t r = 0; r < R; ++r) {
8314 result += fmt::format("{}{}", rindent, lbrace);
8315 rindent = std::string(prefix.size() + 1, ' ');
8316 std::string csep = "";
8317 for (size_t c = 0; c < C; ++c) {
8318 result += fmt::format(
8319 "{}{:>{}}", csep, detail::entry_repr(x(r, c)), col_width);
8320 csep = ", ";

8323 if (r != R - 1) {
8324 result += ",\n";
