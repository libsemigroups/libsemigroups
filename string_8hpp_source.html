51 // A << function for vectors
52 template <typename T>
53 std::ostream& operator<<(std::ostream& os, std::vector<T> const& vec) {
54 if (vec.empty()) {
55 os << "{}";
56 return os;
57 }
58 os << "{{"; // {{ is an escaped single { for fmt
59 for (auto it = vec.cbegin(); it < vec.cend() - 1; ++it) {
60 os << detail::to_string(*it) << ", ";
61 }
62 os << detail::to_string(*(vec.cend() - 1)) << "}}";
63 return os;
64 }

135 // Replace [it1_begin .. it1_begin + (it2_end - it2_begin)] by
136 // [it2_begin .. it2_end], no checks performed.
137 static inline void string_replace(std::string::iterator it1_begin,
138 std::string::const_iterator it2_begin,
139 std::string::const_iterator it2_end) {
140 while (it2_begin < it2_end) {
141 *it1_begin = *it2_begin;
142 ++it1_begin;
143 ++it2_begin;
144 }

147 // Returns true if [first_prefix, last_prefix) is a prefix of [first_word,
148 // last_word).
149 // Not noexcept, because std::equal isn't
150 template <typename S, typename T>
151 bool is_prefix(S first_word, S last_word, T first_prefix, T last_prefix) {
152 LIBSEMIGROUPS_ASSERT(first_word <= last_word);
153 // Check if [first_prefix, last_prefix) equals [first_word, first_word +
154 // (last_prefix - first_prefix))
155 if (!(first_prefix <= last_prefix)) {
156 return false;
157 } else if (!(last_prefix - first_prefix <= last_word - first_word)) {
158 return false;
159 } else if (!std::equal(first_prefix, last_prefix, first_word)) {
160 return false;
161 }
162 return true;
163 }

176 template <typename S, typename T>
177 std::pair<S, T> maximum_common_prefix(S first_word1,
178 S last_word1,
179 T first_word2,
180 T last_word2) {
181 // TODO there must be a better way of doing the static_casts in the next
182 // line
183 while (first_word1 < last_word1 && first_word2 < last_word2
184 && static_cast<uint64_t>(*first_word1)
185 == static_cast<uint64_t>(*first_word2)) {
186 ++first_word1;
187 ++first_word2;
188 }
189 return std::make_pair(first_word1, first_word2);
190 }

192 template <typename T>
193 std::pair<T, T> maximum_common_suffix(T first_word1,
194 T last_word1,
195 T first_word2,
196 T last_word2) {
197 auto p = maximum_common_prefix(std::make_reverse_iterator(last_word1),
198 std::make_reverse_iterator(first_word1),
199 std::make_reverse_iterator(last_word2),
200 std::make_reverse_iterator(first_word2));
201 return std::make_pair(p.first.base(), p.second.base());
202 }

223 // Returns true if [first_suffix, last_suffix) is a suffix of [first_word,
224 // last_word).
225 template <typename S, typename T>
226 bool is_suffix(S first_word, S last_word, T first_suffix, T last_suffix) {
227 return is_prefix(std::make_reverse_iterator(last_word),
228 std::make_reverse_iterator(first_word),
229 std::make_reverse_iterator(last_suffix),
