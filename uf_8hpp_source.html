110 // Union-Find
112
113 // not noexcept because std::array::operator[] and
114 // std::vector::operator[] aren't
115 index_type find(index_type x) const {
116 LIBSEMIGROUPS_ASSERT(x < _data.size());
117 while (true) {
118 index_type y = index(_data[x]);
119 LIBSEMIGROUPS_ASSERT(0 <= y && y < _data.size());
120 if (y == x) {
121 return y;
122 }
123 index_type z = index(_data[y]);
124 LIBSEMIGROUPS_ASSERT(0 <= z && z < _data.size());
125 if (y == z) {
126 return y;
127 }
128 _data[x] = node(z) | rank(_data[x]);
129 x = z;
130 }
131 }
132
133 // not noexcept because UF::find isn't
134 void unite(index_type x, index_type y) {
135 LIBSEMIGROUPS_ASSERT(x < _data.size());
136 LIBSEMIGROUPS_ASSERT(y < _data.size());
137 x = find(x);
138 y = find(y);
139 if (x == y) {
140 return;
141 }
142 rank_type rx = rank(_data[x]);
143 rank_type ry = rank(_data[y]);
144 if (rx > ry) {
145 _data[y] = node(x) | ry;
146 } else {
147 _data[x] = node(y) | rx;
148 if (rx == ry) {
149 _data[y] = node(y) | (ry + 1);
150 }
151 }
152 }

168 // not noexcept because find isn't
169 // if this contains that
170 bool contains(UF const& that) const {
171 LIBSEMIGROUPS_ASSERT(size() == that.size());
172 // reps of that to reps of this
173 std::unordered_map<index_type, index_type> lookup;
174
175 for (index_type i = 0; i < size(); ++i) {
176 index_type j = that.find(i);
177 auto it = lookup.find(j);
178 if (it == lookup.end()) {
179 lookup.emplace(j, find(i));
180 } else if (it->second != find(i)) {
181 return false;
182 }
183 }
184 return true;
185 }

214 // Modifiers
216
217 // not noexcept because std::array::operator[] and
218 // std::vector::operator[] aren't
219 void compress() {
220 for (index_type i = 0; i < _data.size(); ++i) {
221 index_type j = find(index(_data[i]));
222 rank_type rk = (j == i ? 0 : 1);
223 _data[i] = node(j) | rk;
224 }
225 }
226
227 // not noexcept because compress isn't
228 void normalize() {
229 compress();
230 // old -> new class indexing
231 std::unordered_map<index_type, index_type> lookup;
232 container_type data;
233 set_data_capacity(data, size());
234 for (index_type i = 0; i < size(); ++i) {
235 index_type j = find(i);
236 auto it = lookup.find(j);
237 if (it == lookup.end()) {
238 lookup.emplace(j, i);
239 data[i] = node(i); // rank 0
240 } else {
241 data[i] = node(it->second) | 1;
242 }
243 }
244 std::swap(_data, data);
245 }
246
247 // not noexcept because unite isn't
248 void join(UF const& that) {
249 LIBSEMIGROUPS_ASSERT(size() == that.size());
250 for (index_type i = 0; i < _data.size(); ++i) {
251 unite(index(_data[i]), index(that._data[i]));
252 }
253 }
254
255 void swap(UF& that) noexcept {
256 std::swap(_data, that._data);
257 }

450 // Dynamic Union-Find
451 template <typename T = size_t>
452 class Duf : public UF<std::vector<T>> {
453 public:
454 using UF<std::vector<T>>::UF;
455 using size_type = typename UF<std::vector<T>>::size_type;
456 // not noexcept because std::vector::resize can throw
457 void resize(size_type N) {
458 size_type const M = this->size();
459 if (N <= M) {
460 return;
461 }
462 this->_data.resize(N);
463 std::iota(this->_data.begin() + M, this->_data.end(), M);
464 this->init_data(this->_data.begin() + M, this->_data.end());
465 }
466 };
