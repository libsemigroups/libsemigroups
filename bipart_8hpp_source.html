182 template <typename BipartitionOrBlocks, typename Scalar>
183 static void validate_args(std::vector<std::vector<Scalar>> const& blocks) {
184 static_assert(std::is_same_v<BipartitionOrBlocks, Bipartition>
185 || std::is_same_v<BipartitionOrBlocks, Blocks>);
186 static_assert(std::is_signed<Scalar>::value,
187 "the 2nd template parameter Scalar must be signed");
188 int32_t offset = 2;
189 if (!IsBipartition<BipartitionOrBlocks>) {
190 offset = 1;
191 }
192 int32_t m = 0;
193 int32_t deg = 0;
194 std::unordered_set<Scalar> vals;
195 for (size_t i = 0; i < blocks.size(); ++i) {
196 auto const& block = blocks[i];
197 if (block.empty()) {
198 LIBSEMIGROUPS_EXCEPTION("the argument (blocks) is invalid, "
199 "expected all blocks to be non-empty, but "
200 "found empty block in position {}",
201 i);
202 }
203 bool positive = block[0] >= 0;
204
205 for (size_t j = 0; j < block.size(); ++j) {
206 auto x = block[j];
207 vals.insert(x);
208 if (x == 0) {
209 LIBSEMIGROUPS_EXCEPTION("the argument (blocks) is invalid, "
210 "expected non-zero values but found 0 in "
211 "position {} of the block with index {}",
212 j,
213 i);
214 } else if (!IsBipartition<BipartitionOrBlocks> && positive && x < 0) {
215 LIBSEMIGROUPS_EXCEPTION(
216 "the argument (blocks) is invalid, expected every value in the "
217 "block with index {} to be {}tive, but found {} in position {}",
218 i,
219 positive ? "posi" : "nega",
220 x,
221 j);
222 }
223 x = std::abs(x);
224
225 m = std::max(x, m);
226 deg++;
227 }
228 }
229
230 if (m >= static_cast<int32_t>(0x40000000)) {
231 LIBSEMIGROUPS_EXCEPTION(
232 "too many points, expected at most {}, found {}", 0x40000000, m);
233 } else if (deg != offset * m || vals.size() != size_t(deg)) {
234 std::string range, prefix;
235 if (IsBipartition<BipartitionOrBlocks>) {
236 prefix = "the union of";
237 range = fmt::format("{{{}, ..., -1, 1, ..., {}}}", -m, m);
238 } else {
239 prefix
240 = "the set consisting of the absolute values of the entries in ";
241 range = fmt::format("[1, {}]", m);
242 }
243 LIBSEMIGROUPS_EXCEPTION("{} the given blocks is not {},"
244 " only {} values were given",
245 prefix,
246 range,
247 deg);
248 }
249 }
