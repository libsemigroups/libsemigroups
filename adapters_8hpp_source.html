683 template <typename Element,

686 struct OnTuples {
687 static_assert(std::is_same<typename Container::value_type, Point>::value,
688 "the 3rd template parameter is not a container of objects of "
689 "type the 2nd template parameter");
708 void operator()(Container& res,
709 Container const& pt,
710 Element const& x) const {
711 for (size_t i = 0; i < pt.size(); ++i) {
712 ImageRightAction<Element, Point>()(res[i], pt[i], x);
713 }
714 }
