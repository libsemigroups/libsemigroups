559 template <typename Thing>
560 Node& add_node(Thing&& thing) {
561 auto name_str = detail::dot_to_string(std::forward<Thing>(thing));
562 auto [it, inserted] = _nodes.emplace(name_str, Node(name_str));
563 if (!inserted) {
564 LIBSEMIGROUPS_EXCEPTION("there is already a node named {}!", name_str);
565 }
566 return it->second;
567 }
568
569 template <typename Thing>
570 Node& node(Thing&& thing) {
571 auto name_str = detail::dot_to_string(std::forward<Thing>(thing));
572 auto it = _nodes.find(name_str);
573 if (it == _nodes.cend()) {
574 LIBSEMIGROUPS_EXCEPTION("there is no node named {}!", name_str);
575 }
576 return it->second;
577 }

596 template <typename Thing1, typename Thing2>
597 Edge& add_edge(Thing1&& head, Thing2&& tail) {
598 auto head_str = detail::dot_to_string(std::forward<Thing1>(head));
599 auto tail_str = detail::dot_to_string(std::forward<Thing2>(tail));
600 throw_if_not_node(head_str);
601 throw_if_not_node(tail_str);
602 _edges.emplace_back(head_str, tail_str);
603 return _edges.back();
604 }
605
626 template <typename Thing1, typename Thing2>
627 Edge& edge(Thing1&& head, Thing2&& tail) {
628 auto head_str = detail::dot_to_string(std::forward<Thing1>(head));
629 auto tail_str = detail::dot_to_string(std::forward<Thing2>(tail));
630 throw_if_not_node(head_str);
631 throw_if_not_node(tail_str);
632 auto it = std::find_if(_edges.begin(), _edges.end(), [&](Edge const& e) {
633 return e.head == head_str && e.tail == tail_str;
634 });
635 if (it == _edges.cend()) {
636 LIBSEMIGROUPS_EXCEPTION(
637 "there is no edges from {} to {}!", head_str, tail_str);
638 }
639 return *it;
640 }
