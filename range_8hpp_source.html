27 namespace libsemigroups {
28  template <typename TIntegralType, size_t TBegin = 0, size_t TEnd = 0>
29  class IntegralRange {
30  static_assert(
31  std::is_integral<TIntegralType>::value,
32  "the template parameter TIntegralType must be an integral type");
33 
34  public:
35  using value_type = TIntegralType;
36 
37  IntegralRange() : _begin(TBegin), _end(TEnd) {}
38 
39  IntegralRange(TIntegralType begin, TIntegralType end)
40  : _begin(begin), _end(end) {}
41 
42  class iterator {
43  friend class IntegralRange;
44 
45  public:
46  TIntegralType operator*() const {
47  return _i;
48  }
49 
50  iterator const& operator++() {
51  ++_i;
52  return *this;
53  }
54 
55  iterator operator++(int) {
56  iterator copy(*this);
57  ++_i;
58  return copy;
59  }
60 
61  bool operator==(iterator const& other) const {
62  return _i == other._i;
63  }
64 
65  bool operator!=(iterator const& other) const {
66  return _i != other._i;
67  }
68 
69  bool operator<(iterator const& other) const {
70  return _i < other._i;
71  }
72 
73  protected:
74  explicit iterator(TIntegralType start) : _i(start) {}
75 
76  private:
77  TIntegralType _i;
78  };
