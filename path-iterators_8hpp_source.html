128 // Note that while the complexity of this is bad, it repeatedly does depth
129 // first searches, and so will examine every node and edge of the graph
130 // multiple times (if u -a-> v belongs to a path of length 10, then it will
131 // be traversed 10 times). But the performance of this iterator is
132 // dominated by memory allocation (when creating
133 // iterators, at least), and so this doesn't seem that bad.
134 template <typename Node>
135 class const_pislo_iterator {
136 public:
137 using node_type = Node;
138 using value_type = word_type;
139 using size_type = typename std::vector<value_type>::size_type;
140 using difference_type = typename std::vector<value_type>::difference_type;
141 using const_pointer = typename std::vector<value_type>::const_pointer;
142 using pointer = typename std::vector<value_type>::pointer;
143 using const_reference = typename std::vector<value_type>::const_reference;
144 using reference = const_reference;
145 using iterator_category = std::forward_iterator_tag;
146
147 private:
148 size_type _length;
149 detail::const_pilo_iterator<Node> _it;
150 size_type _max;
151 node_type _source;
152
153 public:
154 // None of the constructors are noexcept because the corresponding
155 // constructors for detail::const_pilo_iterator are not.
156
157 const_pislo_iterator();
158
159 const_pislo_iterator(const_pislo_iterator const&);

161 const_pislo_iterator(const_pislo_iterator&&);

173 // noexcept because comparison of detail::const_pilo_iterator is noexcept
174 [[nodiscard]] bool
175 operator==(const_pislo_iterator const& that) const noexcept {
176 return _length == that._length && _it == that._it;
177 }
178
179 // noexcept because operator== is noexcept
180 [[nodiscard]] bool
181 operator!=(const_pislo_iterator const& that) const noexcept {
182 return !(this->operator==(that));
183 }
184
185 [[nodiscard]] const_reference operator*() const noexcept {
186 return *_it;
187 }
188
189 [[nodiscard]] const_pointer operator->() const noexcept {
190 return &(*_it);
191 }
192
193 [[nodiscard]] node_type target() const noexcept {
194 if (_it == cend_pilo(_it.word_graph()) && _length == UNDEFINED) {
195 return UNDEFINED;
196 }
197 return _it.target();
198 }
199
200 // prefix - not noexcept because cbegin_pilo isn't
201 const_pislo_iterator const& operator++();
202
203 // postfix - not noexcept because copy constructor isn't
204 const_pislo_iterator operator++(int) {
205 return detail::default_postfix_increment<const_pislo_iterator>(*this);
206 }

208 void swap(const_pislo_iterator& that) noexcept;
209 };
