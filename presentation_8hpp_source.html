621 template <typename Word>
622 void throw_if_not_normalized(Presentation<Word> const& p,
623 std::string_view arg = "1st") {
624 auto first = std::begin(p.alphabet()), last = std::end(p.alphabet());
625 if (!std::is_sorted(first, last)) {
626 LIBSEMIGROUPS_EXCEPTION("the {} argument (presentation) must have "
627 "sorted alphabet, found {}",
628 arg,
629 p.alphabet());
630 }
631
632 auto it = std::max_element(first, last);
633 if (it != last && *it != p.alphabet().size() - 1) {
634 LIBSEMIGROUPS_EXCEPTION("the {} argument (presentation) has invalid "
635 "alphabet, expected [0, ..., {}] found {}",
636 arg,
637 p.alphabet().size() - 1,
638 p.alphabet());
639 }
640 }
